def each
  case self
  when AndNode, OrNode
    @children.first.each{|c| yield(c)}
    yield(self)
    @children.last.each{|c| yield(c)}
  when NotNode
    yield(self)
    @children.first.each{|c| yield(c)}
  when VarNode
    yield(self)
  end
end
